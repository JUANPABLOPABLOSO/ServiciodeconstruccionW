    <h2>¡Adivina el número!</h2>
    <p>Adivina un número entre 1 y 100.</p>
    <input type="number" id="guessInput">
    <button onclick="checkGuess()">¡Adivinar!</button>
    <p id="result"></p>
  </div>

  <script>
    // Generar un número aleatorio entre 1 y 100
    var targetNumber = Math.floor(Math.random() * 100) + 1;
    // Contador de intentos
    var attempts = 0;

    // Función para comprobar el intento del usuario
    function checkGuess() {
      var guess = parseInt(document.getElementById('guessInput').value);
      if (isNaN(guess) || guess < 1 || guess > 100) {
        alert('Por favor, introduce un número válido entre 1 y 100.');
        return;
      }

      attempts++;
      if (guess === targetNumber) {
        document.getElementById('result').innerHTML = `¡Felicidades! ¡Has adivinado el número en ${attempts} intentos!`;
        document.getElementById('guessInput').disabled = true; // Deshabilitar el campo de entrada
        document.querySelector('button').disabled = true; // Deshabilitar el botón de adivinar
      } else if (guess < targetNumber) {
        document.getElementById('result').innerHTML = 'Intenta con un número mayor.';
      } else {
        document.getElementById('result').innerHTML = 'Intenta con un número menor.';
      }

      // Limpiar el campo de entrada
      document.getElementById('guessInput').value = '';

      // Si se han agotado los intentos, mostrar el número correcto
      if (attempts >= 5 && guess !== targetNumber) {
        document.getElementById('result').innerHTML = `¡Lo siento! Se han agotado los intentos. El número correcto era ${targetNumber}.`;
        document.getElementById('guessInput').disabled = true; // Deshabilitar el campo de entrada
        document.querySelector('button').disabled = true; // Deshabilitar el botón de adivinar
      }
    }
  </script>
</body>

  <script>
    var towerHeight = 0;

    // Función para agregar un bloque a la torre
    function addBlock() {
      towerHeight++;
      var block = document.createElement('div');
      block.classList.add('block');
      block.style.height = (towerHeight * 20) + 'px';
      document.getElementById('tower').appendChild(block);
      
      // Verificar si la torre se desequilibra
      if (towerHeight > 5 && Math.random() < 0.3) {
        document.getElementById('result').innerHTML = '¡La torre se desequilibró! ¡Juego terminado!';
        document.querySelector('button').disabled = true; // Deshabilitar el botón después de que la torre se desequilibre
      }
    }
